2. Crea una función que reciba como parámetros: día, mes, año, hora, minutos y segundos y cree una fecha con esos datos. Si la función no recibe ningún parámetro creará la fecha actual. La función mostrará un alert con la información en el siguiente formato: “Hoy es lunes, 1 de noviembre de 2021 y son las 17 horas, 15 minutos y 38 segundos.”:


                            function formatearFechaYHora() {
                            // Recibo los valores que introduce el usuario en el formulario
                            let fechaYHora = document.getElementById("fechaYHora").value;
                            // El formato esperado es 'YYYY/MM/DD HH:mm:ss'
                            const formatoFechaHora = /^(\d{4})\/(\d{2})\/(\d{2}) (\d{2}):(\d{2}):(\d{2})$/;

                            // Valido si el valor cumple con el formato
                            if (!formatoFechaHora.test(fechaYHora)) {
                              fechaYHoraObj = new Date(); // Si no es válido, mostrar la fecha y hora actual
                            } else {
                              fechaYHoraObj = new Date(fechaYHora);
                            }

                            function formatearFechaConMensaje(fechaYHoraObj) {
                              const diasSemana = [
                                "Domingo",
                                "Lunes",
                                "Martes",
                                "Miércoles",
                                "Jueves",
                                "Viernes",
                                "Sábado",
                              ];
                              const nombresMeses = [
                                "enero",
                                "febrero",
                                "marzo",
                                "abril",
                                "mayo",
                                "junio",
                                "julio",
                                "agosto",
                                "septiembre",
                                "octubre",
                                "noviembre",
                                "diciembre",
                              ];

                              const diaSemana = diasSemana[fechaYHoraObj.getDay()];
                              const dia = fechaYHoraObj.getDate();
                              const mes = nombresMeses[fechaYHoraObj.getMonth()];
                              const año = fechaYHoraObj.getFullYear();
                              const horas = fechaYHoraObj.getHours();
                              const minutos = fechaYHoraObj.getMinutes();
                              const segundos = fechaYHoraObj.getSeconds();

                              // Y la muestro formateada por consola
                              return `< div>Hoy es ${diaSemana}, ${dia} de ${mes} de ${año} y son las ${horas} horas, ${minutos} minutos y ${segundos} segundos.< /div>`;
                            }

                            // Obtengo el elemento donde mostrar el resultado
                            const resultado = document.getElementById("resultado");

                            // Llamo a la función que formatea la fecha con mensaje
                            resultado.innerHTML = formatearFechaConMensaje(fechaYHoraObj);
